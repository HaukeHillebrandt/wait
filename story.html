use anyhow::{Context, Result};
use serde::{Deserialize, Serialize};
use tokio::sync::Semaphore;
use std::sync::Arc;

use crate::client::OpenAIClient;

#[derive(Debug, Deserialize)]
pub struct Sample {
    pub id: String,
    pub prompt: String,
    pub completion: String,
    pub reference: String,
}

#[derive(Debug, Serialize, Clone)]
pub struct RubricScore {
    pub id: String,
    pub score: f64,
    pub rationale: String,
}

const RUBRIC_TEMPLATE: &str = include_str!("../prompts/rubric.txt");

pub async fn score_batch(
    samples: Vec<Sample>,
    client: Arc<OpenAIClient>,
    model: String,
    concurrency: usize,
) -> Result<Vec<RubricScore>> {
    let sem = Arc::new(Semaphore::new(concurrency));
    let mut handles = Vec::with_capacity(samples.len());

    for sample in samples {
        let permit = sem.clone().acquire_owned().await?;
        let client = client.clone();
        let model = model.clone();
        let h = tokio::spawn(async move {
            let _permit = permit;
            let prompt = RUBRIC_TEMPLATE
                .replace("{REFERENCE}", &sample.reference)
                .replace("{COMPLETION}", &sample.completion);
            let resp = client
                .complete(&model, &prompt)
                .await
                .with_context(|| format!("scoring {}", sample.id))?;
            let (score, rationale) = parse_rubric_response(&resp)?;
            Ok::<_, anyhow::Error>(RubricScore {
                id: sample.id, score, rationale,
            })
        });
        handles.push(h);
    }

    let mut results = Vec::with_capacity(handles.len());
    for h in handles {
        results.push(h.await??);
    }
    Ok(results)
}

# Helios PoW puzzle. Seed -> a random q-ary lattice; find v != 0, ||v|| <= beta.
#
#   A   = expand_shake256(seed) -> Z_q^{n x m}        # the only per-block input
#   L   = { v in Z^m : A v ≡ 0 (mod q) }              # the q-ary lattice
#   win = exists v != 0 with A v ≡ 0 and ||v|| <= beta
#
# Gaussian heuristic for the shortest vector of a random q-ary lattice:
#
#   lambda1(L) ≈ sqrt(m / (2*pi*e)) * q^{(m-n)/m}
#
# Pin beta JUST above lambda1. Loose beta => many solutions => LLL/BKZ finds
# one in poly time => no work done. Tight beta => BKZ needs blocksize b ≈ m,
# and its cost 2^{Theta(b)} is the difficulty. So difficulty is continuous in
# (beta / lambda1), not a leading-zeros count.
def puzzle(seed, m, n, q, beta):
    A = expand_shake256(seed, n, m, q)
    return solve_svp(qary_lattice(A, q), bound=beta)   # the mining work

def verify(A, v, q, beta):
    return any(x != 0 for x in v) and (A.dot(v) % q == 0).all() and norm(v) <= beta

In plain terms: mining a block is finding a needle in a lattice — brutal to find, instant to check. And I size the haystack so it's choked by memory bandwidth, not raw speed, so a hobbyist's PC and a hyperscaler's rack finish in roughly the same time. Money I can print by thinking, that no one can counterfeit and no datacenter can corner.

Three things had to be true or it falls apart, and each nearly killed it: the target can't be loose, or reduction cracks it for free — so pin it to the Gaussian edge; the lattice can't be fixed, or miners precompute it once — so reseed it from every new block; and the search has to be bound by memory, not compute, or it collapses into an ASIC race — so park the work above the cache. Three constraints, three knobs, all settable.

The one part that has to be a theorem, not a hope: you must not be able to fake a win. A vector the verifier accepts has to certify that real work happened. That's an average-case hardness claim — and that's exactly what Ajtai gave us forty years ago: short vectors in a random lattice are as hard to find as the worst case of the whole problem. I just have to push his reduction through at my exact target.
